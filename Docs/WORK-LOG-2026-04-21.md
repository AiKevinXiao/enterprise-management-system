```javascript
// 权限码校验中间件
function permissionMiddleware(code) {
    return (req, res, next) => {
        if (!req.user || !req.user.permissions?.includes(code)) {
            return res.status(403).json({ error: '权限不足' });
        }
        next();
    };
}

// 数据范围注入中间件
function dataScopeMiddleware() {
    return (req, res, next) => {
        req.dataScope = {
            mode: req.user.data_scope,   // 'all' | 'dept' | 'self'
            deptIds: getSubDeptIds(req.user.dept_id),
            userId: req.user.user_id
        };
        next();
    };
}
```

## 二、前端权限控制

### 1. PermissionManager 工具函数

**文件**：`frontend/common.js`

在 `AuthManager` 中添加：

```javascript
class PermissionManager {
    constructor() {
        this.permissions = [];
    }
    init(userData) {
        this.permissions = userData.permissions || [];
    }
    hasPermission(code) {
        return this.permissions.includes(code);
    }
    hasAnyPermission(codes) {
        return codes.some(c => this.permissions.includes(c));
    }
}

window.hasPermission = (code) => PermissionManager.instance.hasPermission(code);

function initPermissions() {
    const user = JSON.parse(localStorage.getItem('ems_user'));
    if (user?.data) PermissionManager.instance.init(user.data);
    // 扫描所有 data-permission 属性，无权限则隐藏
    document.querySelectorAll('[data-permission]').forEach(el => {
        if (!hasPermission(el.dataset.permission)) el.style.display = 'none';
    });
}
```

`DOMContentLoaded` 时调用 `initPermissions()`，页面加载后自动隐藏无权限按钮。
